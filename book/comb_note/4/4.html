基于枚举的生成算法

算法叙述

给定一个整数，对其赋予方向， ，如果一个整数的箭头指向一个与其相邻但比它小的的整数，那么这个整数是可移动的。

从开始。

  1. 找到最大的可移动整数，如果没有结束算法
  2. 交换它和箭头所指向的数
  3. 改变所有满足的整数上的箭头的方向。

c++ code

#include <bits/stdc++.h>

using namespace std;

void prt(int *arr,int len) {
    for (int i = 0; i < len; i++) {
        printf("%d ",abs(arr[i]));
    }
    printf("\n");
}

bool nxt_a(int *arr,int len) {
    int id,cmp;
    int ret = 0;
    int ai;
    for (int i = 0; i < len; i++) {
        ai = abs(arr[i]);

        id = i + (arr[i]>0 ? -1 : 1);
        if (0<=id && id<len) cmp = arr[id];
        else cmp = len + 1;

        if (abs(cmp) < ai)
            if (ret < ai)
                ret = ai;
    }

    if (ret == 0) return false;

    for (int i = 0; i< len; i++) {
        ai = abs(arr[i]);
        if (ai == ret) {
            id = i + (arr[i]>0 ? -1 : 1);
            swap (arr[i] , arr[id]);
            break;
        }
    }

    for (int i = 0; i< len; i++)
        if (abs(arr[i]) > ret) arr[i] = -arr[i];

    return true;
}

int main(){
    // pos to left, neg to right
    int a[10]={1,2,3,4,5};

    int n = 5;
    do{
        prt(a,n);
    }while ( nxt_a(a,n) );
    return 0;
}
